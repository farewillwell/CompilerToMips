.data

.text

# enter main
jal main


# leave main
j END

main:
mainb0:

# br label %b1
j mainb1

mainb1:

# %V3 = icmp slt i32 %V9, 100
li $t1, 100
slt $v1, $t3, $t1

# br i1 %V3, label %b3, label %b4
bne $v1, $zero, mainb3
beq $v1, $zero, mainb4
mainb3:

# br label %b1
j mainb1

mainb4:

# call void @putint(i32 0)
li $a0, 0
li $v0, 1
syscall

# ret i32 0
li $v0, 0
jr $ra

END:
li $v0, 10
syscall
