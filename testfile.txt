int getDif3N(){
    int i,j,k;
    int cnt=0;
    for(i=1;i<=3;i=i+1){
        for(j=1;j<=3;j=j+1){
            for(k=1;k<=3;k=k+1){
                if(i!=j&&i!=k&&j!=k){
                    cnt=cnt+1;
                    printf("%d\n",cnt);
                }
            }
        }
    }
    return cnt;
}

int main(){
    printf("%d",getDif3N());
    return 0;
}
